class DataBase:
    __instance = None

    def __new__(cls, *args, **kwargs):
        if cls.__instance is None:
            cls.__instance = super().__new__(cls)

        return cls.__instance

    def __del__(self):
        DataBase.__instance = None

    def __init__(self, user, psw, port):
        self.user = user
        self.psw = psw
        self.port = port

    def connect(self):
        print(f'соединение с БД: {self.user}, {self.psw}, {self.port}')

    def close(self):
        print('закрытие соединения с БД')

    def read(self):
        return 'данные из БД'

    def write(self,data):
        print(f'запись в БД {data}')


db = DataBase('root', '1234', 80)
db2 = DataBase('root2', '5678', 40)
print(id(db), id(db2))
db.connect()
db2.connect()

@Grigorii_Tashu, потому что через встроенный __new__ можно гарантировать на 100% без любых условностей отслеживание всех ситуаций создания экземпляров данного класса с полным контролем этих ситуаций. Любой другой метод организации такой же логики требует овер-инджиниринга (дублирование ф-нкционала в данном случае) для обеспечения того же самого.
Я могу вас уверить, что просто создавать 1 экземпляр и ввести условность больше не создавать их для данного класса не даёт никакой гарантии, поэтому этот паттерн и был принят на вооружение. Как говориться правило написано "кровью". И да, свою я тоже на этом поприще проливал.

Сборщик мусора тут работает по такому же принципу как и везде для любого изменяемого типа данных в Python - когда количество ссылок на объект становиться 0, сборщик мусора его уничтожает.

Т.к. потеря единственной ссылки в атрибуте класса автоматически не организована, то мы делаем это руками.

Пока представте, что __new__ это своеобразный декоратор для __init__. Это не совсем так (если честно, то вообще не так), но для начального понимания как идёт передача аргументов этого хватит. Аргументы сначала попадают в __new__, включая cls (Какой класс нужно создать).

Ниже приведён пример того, как это работает.
Эти две строки равнозначны
person = object.__new__(Person, 'John')
person.__init__('John')

Здесь в первом случае словарь будет пустой, т.к. объект ещё не создан.
person = object.__new__(Person, 'John')
print(person.__dict__)

Здесь в словаре будет {'name': 'John'}
person.__init__('John')
print(person.__dict__)

Общая картина:

class Person:

    def __new__(cls, name):
        print(f'Creating a new {cls.__name__} object...')
        obj = object.__new__(cls)
        return obj

    def __init__(self, name):
        print(f'Initializing the person object...')
        self.name = name


person = Person('John')

Консоль:
    Creating a new Person object...
    Initializing the person object...

@Алексей_Лысов, если вы какую-то логику и придумываете в своей проге, то мб вы и будете использовать как-то экземпляр в каком-то качестве. Но логика такая - раз удалили сущность в проге (экземпляр одиночку в данному случае), значит нужно и то что её реализует грохнуть.

Мы не используем какой-то заранее подготовленный для этого всего случая функционал, его нету.

@Алексей_Лысов,  Атрибут __instance относится к пространству имен класса. Это атрибут класса а не объекта, а сборщик мусора освобождает память, занимаемую объектами а не ломает классы.... Более того,  атрибут __instance содержит ссылку на созданный объект, а значит сборщик мусора этот объект не тронет( пока существует ссылка на него). Поэтому надо явно "обнулять" ссылку при удалении объекта, для этого и переопределяется метод __del__ - само это не произойдет))

# Реализация Singletona
class Data:
    __in = None
    def __new__(cls, *args, **kwargs):
        if cls.__in == None:  # то есть ЭК ещё ни разу не создавался !
            cls.__in = super().__new__(cls)  # Наследование от базового класса Object,  super().__new__(cls) 
        return cls.__in # А если он уже был, тогда - просто возвращается прежнее super().__new__(cls) 

    def __init__(self, name = "Cary"):  # При создании ЭК ему передаётся один аргумент, или может не передаваться ничего
        self.name = name

    def get_name(self): # Функция показывает значение атрибута name
        print(self.name)
    def get_in(self):
        print(self.__in)
    def set_in(self, inw):
        self.__in = inw

d = Data("Coca")  # Передаю имя Соса
d.get_name()   # Печатаю его
d2 = Data()    #  Создаю ЭК, не передавая аргумента
d2.get_name()  #  По умолчанию Cary
Data.get_name(d)  # снова выводится Cary

Data.get_name(d)   # Печатаю name ЭК -  d ( a не d3) = Gucha

d.set_in("jjj")  # устанавливаю новое значение свойства __in
d.get_in()  # Печатается jjj
d2.get_in()  #  Печатается jjj
