# setattr(class, str(variable), значение)
#
# Задаёт значение для переменной в классе

# 1. Имя класса

# 3. Значение переменной


# getattr(class, str(variable), Значение которое будет возвращаться при ошибке(не обяз.))
#
# Возврощает значение переменной в указанном классе
# Если оно отсутсвует то возвращает ошибку или третий аргумент
# 1. Имя класса
# 2. Имя переменной в виде строки


# hasattr(class, str(variable))
#
# Проверяет переменную в указанном классе на наличие
# Если оно отсутствует возврощает False в противном случае True
# 1. Имя класса
# 2. Имя переменной в виде строки

#
# Удаляет переменную в указанном классе
# 1. Имя класса
